//id.l


%{
  #include "y.tab.h"
%}

%%
[a-zA-Z_][a-zA-Z_0-9]*   return letter; 
[0-9]                    return digit; 
[ \t]                    ;      // ignore spaces/tabs
\n                       return 0;
.                        return yytext[0]; 
%%

int yywrap()
{ 
  return 1; 
}


//id.y


%{
   #include<stdio.h>
   #include<stdlib.h>
   
   int yyerror(const char *msg);
   int yylex(void);
   int valid = 1;
%}

%token digit letter

%%

start : letter s   /* must start with a letter */
      ;

s : letter s       /* followed by letters */
  | digit s        /* or digits */
  |                /* empty */
  ;

%%

int yyerror(const char *msg)
{
    printf("\nIt is not an identifier!\n");
    valid = 0;
    return 0;
}

int main()
{
    printf("\nEnter a name to test for identifier: ");
    yyparse();
    if(valid)
    {
        printf("\nIt is an identifier!\n");
    }
    return 0;
}

Execution:
yacc -d id.y
lex id.l
gcc y.tab.c lex.yy.c -o identifier -lfl
./identifier
